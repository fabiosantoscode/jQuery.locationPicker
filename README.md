jQuery.locationPicker()
===

## Example

    ...
    <input name="loc" id="loc">

    <script src="http://ajax.googleapis.com/ajax/libs/jquery/1.6.2/jquery.min.js"></script>
    <script src="js/jquery.location-picker.js"></script>
    <script>
    jQuery(document).ready(function ($) {
      $('#loc').locationPicker();
    });
    </script>
    ...
